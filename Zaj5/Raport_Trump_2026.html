text <- readLines(file.choose(), encoding = "UTF-8")

data(DictionaryGI)

df_GI <- data.frame(index = seq_along(sentimentGI),

                    Dictionary = "GI")

df_GI <- na.omit(df_GI)

ggplot(df_GI, aes(x = value)) +
  geom_bar(fill = "green", alpha = 0.7) +

Sentyment w czasie

df_time <- data.frame(
  sentence = 1:length(sentiment[,1]),
  GI = sentiment$SentimentGI,
  HE = sentiment$SentimentHE,
  LM = sentiment$SentimentLM,
  QDAP = sentiment$SentimentQDAP
)

df_time <- df_time[complete.cases(df_time), ]

ggplot(df_time, aes(x = sentence)) +
  geom_line(aes(y = GI), color = "green") +
  geom_line(aes(y = HE), color = "blue") +
  geom_line(aes(y = LM), color = "orange") +
  geom_line(aes(y = QDAP), color = "red") +
  theme_gdocs() +
  labs(title = "Zmiana sentymentu w czasie",
       x = "Zdania",
       y = "Sentyment")
